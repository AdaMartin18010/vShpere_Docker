# 隔离熵计算
def isolation_entropy(leak_matrix):
    """量化隔离强度"""
    leak_probs = leak_matrix[leak_matrix > 0]
    return -np.sum(leak_probs * np.log2(leak_probs))

# 实测结果

# 含义: 隔离熵越小，隔离越强
